  bearerAuth: process.env["MIDDAY_BEARER_AUTH"] ?? "",
});

  const result = await midday.teams.patchV1TeamsId({
    id: "123e4567-e89b-12d3-a456-426614174000",
    requestBody: {
      name: "Acme Corporation",
      email: "team@acme.com",
      logoUrl: "https://cdn.midday.ai/logos/acme-corp.png",
      baseCurrency: "USD",
